(function() { function formatTime(date) { const optionsDate = { day: '2-digit', month: '2-digit', year: 'numeric' }; const optionsTime = { hour: '2-digit', minute: '2-digit', second: '2-digit' }; return date.toLocaleDateString('ru-RU', optionsDate) + ' ' + date.toLocaleTimeString('ru-RU', optionsTime); } function updateTime() { const timeElement = document.getElementById('time'); if (!timeElement) return; const now = new Date(); timeElement.innerHTML = formatTime(now); } document.addEventListener('DOMContentLoaded', function() { updateTime(); setInterval(updateTime, 13000); }); })();
